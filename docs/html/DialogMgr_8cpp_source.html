154  void DialogMgr::update(Game &game)
155  {
156  if (this->waiting()) {
157  this->_waiting = !game.state.settings.input->actionPressed(Input::DIALOG);
158  if (!this->_waiting)
159  this->_displayed.clear();
160  return;
161  }
162  if (this->_left.empty())
163  return;

210  void DialogMgr::_consumeCharacter(Game &game)
211  {
212  char c = this->_left[0];
213 
214  this->_left = this->_left.substr(1);
215  switch (c) {
216  case '\\':
217  this->_displayed += this->_left[0];
218  this->_left = this->_left.substr(1);
219  break;
220  default:
221  this->_displayed += c;
222  break;
223  case '\n':
224  this->_waiting = !this->_displayed.empty();
225  break;
226  case '%':
227  auto elems = this->_parseCommand();
228  auto cmd = elems[0];
229 
230  elems.erase(elems.begin());
231  this->_executeCommand(game, cmd, elems);
232  if (!this->_left.empty())
233  this->_consumeCharacter(game);
234  }
235  this->_waiting |= this->_left.empty() && !this->_displayed.empty();
236  }

238  std::vector<std::string> DialogMgr::_parseCommand()
239  {
240  std::vector<std::string> result;
241  int i = 0;
242  char quote = 0;
243  std::string token;
244  char c = this->_left[i];
245 
246  try {
247  while (c != '%') {
248  switch (c) {
249  case '"':
250  case '\'':
251  if (quote == c) {
252  quote = 0;
253  result.push_back(token);
254  token.clear();
255  } else if (!quote)
256  quote = c;
257  else
258  token.push_back(c);
259  break;
260  case ' ':
261  case '\t':
262  case '\n':
263  if (quote)
264  token.push_back(c);
265  else if (!token.empty()) {
266  result.push_back(token);
267  token.clear();
268  }
269  break;
270  case '\\':
271  token.push_back(this->_left.at(++i));
272  break;
273  default:
274  token.push_back(c);
275  }
276  c = this->_left.at(++i);
277  }
278  } catch (std::out_of_range &) {
279  if (quote)
280  throw UnfinishedStringLiteralException("Unfinished string found (starting here: " + std::string(&quote, 1) + token.substr(0, 6) + ")");
281  throw UnfinishedCommandException("Unfinished command found (starting here: " + this->_left.substr(0, 6) + ")");
282  }
283  result.push_back(token);
284  this->_left = this->_left.substr(i + 1);
285  return result;
286  }

313  this->_holderFocus = !this->_holderFocus;
