104 
105  logger.debug("Name -> " + this->_name);
106 
107  stream >> this->_id;

109 
110  for (stream >> str; str != "THFG_ECS_Entity_End" && !stream.eof(); stream >> str) {
111  logger.debug("Unserialize component " + str);
112  this->_components.emplace_back(Factory::ComponentFactory::build(game, str, stream));
113  }
114 
115  if (str != "THFG_ECS_Entity_End")
116  throw InvalidSerializedString("Unexpected EOF");
