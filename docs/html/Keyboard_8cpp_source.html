186  void Keyboard::_updateState()
187  {
188  for (unsigned i = 0; i < NB_OF_ACTION; i++) {
189  bool press = this->actionPressed(static_cast<Action>(i));
190 
191  if (press != this->_state[i]) {
192  this->_state[i] = press;
193  this->_events.emplace_back(
194  press ? Event::EVENT_TRIGGERED : Event::EVENT_STOPPED_TRIGGER,
195  static_cast<Action>(i)
196  );
197  }
198  }
