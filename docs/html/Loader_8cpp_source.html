41  void Loader::loadAssets(Game &game)
42  {
43  std::ifstream stream{"assets/list.json"};
44  nlohmann::json data;
45 
46  logger.debug("Opening main window");
47  game.resources.screen.reset(new Rendering::Screen{"THFgame"});
48 
49  loadSettings(game);
50 
51  //We do this because on MinGW std::random_device always produce the same output
52  //(also we don't need the most unpredictable seed)
53  game.resources.random.seed(time(nullptr));
54 
55  logger.debug("Opening file assets/list.json");
56  if (stream.fail())
57  throw CorruptedAssetsListException("Cannot open assets list from assets/list.json");
58 
59  try {
60  logger.debug("Parsing json");
61  stream >> data;
62 
63  logger.debug("Loading icon");
64  if (data["icon"].is_null())
65  logger.warn("No Icon is marked for loading");
66  else if (!game.resources.icon.loadFromFile("assets/" + static_cast<std::string>(data["icon"])))
67  logger.error("Cannot load file assets/" + static_cast<std::string>(data["icon"]));
68  else
69  game.resources.screen->setIcon(
70  game.resources.icon.getSize().x,
71  game.resources.icon.getSize().y,
72  game.resources.icon.getPixelsPtr()
73  );
74 
75  logger.debug("Loading musics");
76  loadAssetsFromJson("Musics", data["musics"], game.resources.musics);
77 
78  logger.debug("Loading sfx");
79  loadAssetsFromJson("Sound effects", data["sfx"], game.resources.soundBuffers);
80 
81  logger.debug("Loading sprites");
82  loadAssetsFromJson("Sprites", data["sprites"], game.resources.textures);
83  } catch (nlohmann::detail::parse_error &e) {
84  throw CorruptedAssetsListException("The JSON file has an invalid format: " + std::string(e.what()));
85  } catch (nlohmann::detail::type_error &e) {
