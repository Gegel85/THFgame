100 
101  try {
102  logger.debug("Parsing json");
103  stream >> data;
104  stream.close();
105 
106  logger.debug("Loading icon");
107  if (data["icon"].is_null())
108  logger.warn("No Icon is marked for loading");
109  else if (!game.resources.icon.loadFromFile("assets/" + static_cast<std::string>(data["icon"])))
110  logger.error("Cannot load file assets/" + static_cast<std::string>(data["icon"]));
111  else
112  game.resources.screen->setIcon(
113  game.resources.icon.getSize().x,
114  game.resources.icon.getSize().y,
115  game.resources.icon.getPixelsPtr()
116  );
117 
118  logger.debug("Loading musics");
119  loadAssetsFromJson(game.state.settings, "Musics", data["musics"], game.resources.musics);
120 
121  logger.debug("Loading sfx");
122  loadAssetsFromJson(game.state.settings, "Sound effects", data["sfx"], game.resources.soundBuffers);
123 
124  logger.debug("Loading sprites");
125  loadAssetsFromJson(game.state.settings, "Sprites", data["sprites"], game.resources.textures);
126 
127  logger.debug("Loading items json");
128  loadItems(game);
129  } catch (nlohmann::detail::parse_error &e) {

131  } catch (nlohmann::detail::type_error &e) {
