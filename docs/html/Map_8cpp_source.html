51  this->loadFromFile("assets/maps/map_" + std::to_string(this->_links[Input::RIGHT]) + ".map");
52  pos.x = size.x / -2.;
53  } else if (pos.y + size.y / 2. > this->_size.y * this->_tileSize) {
54  this->loadFromFile("assets/maps/map_" + std::to_string(this->_links[Input::DOWN]) + ".map");
55  pos.y = size.y / -2.;
56  }
57 
58  for (auto &trigger : this->_tpTriggers)
59  if (
60  -this->_tileSize < trigger.location.x && trigger.location.x < static_cast<int>(size.x + this->_tileSize) &&
61  -this->_tileSize < trigger.location.y && trigger.location.y < static_cast<int>(size.y + this->_tileSize)
62  ) {

102  void Map::_loadFromStream(std::istream &stream, bool loadEntities)
103  {
104  char byte;
105 
106  logger.debug("Loading map");
107  do {
108  stream.read(&byte, 1);
109  if (byte)
110  this->_tileMap.push_back(byte);
111  } while (byte);
112  logger.debug("Tilemap file is '" + this->_tileMap + "'");
113 
114  this->_tileSize = _readInteger<unsigned char>(stream);
115  logger.debug("Tilesize is " + std::to_string(this->_tileSize));
116 
117  this->_size.x = _readInteger<unsigned short>(stream);
118  this->_size.y = _readInteger<unsigned short>(stream);
119  logger.debug("Map size is " + toString(this->_size) + " tiles");
120 
121  for (auto len = _readInteger<unsigned>(stream); len != 0; len--) {
122  auto &result = this->_tpTriggers.emplace_back(stream);
123 
124  logger.debug(
125  std::to_string(len) + " teleporters remaining ->"
126  "Position: " + toString(result.location) +
127  "Map: " + std::to_string(result.mapId) +
128  "Spawn: " + toString(result.mapSpawn)
129  );
130  }
131 
132  std::memset(this->_links, 0, sizeof(this->_links));
133  for (unsigned i = 0; i < 4; i++) {
134  this->_links[i] = _readInteger<unsigned short>(stream);
135  logger.debug("Link " + Input::actionToString(static_cast<Input::Action>(i)) + " is to map " + std::to_string(this->_links[i]));
136  }
137 
138  logger.debug("Fetching objects");
139  for (unsigned y = 0; y < this->_size.y; y++)
140  for (unsigned x = 0; x < this->_size.x; x++)
141  this->_objects.push_back(_readInteger<unsigned char>(stream));
142 
143  if (loadEntities) {
144  logger.debug("Loading entities");
145  stream >> this->_core;
146 
147  if (this->_core.getEntityByName("Player").empty())
148  this->_core.makeEntity("Player");
149  }
150  logger.debug("Operation completed");
151  }

182  sf::Vector2f pos = this->_cameraCenter;
183 
184  pos.x -= size.x / 2.;
185  pos.y -= size.y / 2.;
186  _game.resources.screen->setCamera(pos);
187  pos.x /= this->_tileSize;
188  pos.y /= this->_tileSize;
189  size.x = ceil(size.x / static_cast<float>(this->_tileSize));
190  size.y = ceil(size.y / static_cast<float>(this->_tileSize));
191 
192  for (unsigned y = pos.y < 0 ? 0 : pos.y; y < pos.y + size.y && y < this->_size.y; y++) {
193  for (unsigned x = pos.x < 0 ? 0 : pos.x; x < pos.x + size.x && x < this->_size.x; x++) {
194  _game.resources.screen->draw(
195  _game.resources.textures.at(this->_tileMap),
196  {
197  static_cast<float>(x * this->_tileSize),
198  static_cast<float>(y * this->_tileSize)
199  },
200  {
201  static_cast<int>((this->_objects.at(x + y * this->_size.x) & 0x7F) * this->_tileSize),
202  0,
203  static_cast<int>(this->_tileSize),
204  static_cast<int>(this->_tileSize)
