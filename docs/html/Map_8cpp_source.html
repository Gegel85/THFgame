132  void Map::update()
133  {
134  this->_core.update();
135 
136  auto &pos = this->_getPlayerPosition();
137  auto size = this->_getPlayerSize();
138 
139  if (pos.x < size.x / -2.) {
140  this->_serialize("saves/map_" + std::to_string(this->_id) + ".sav");
141  this->_loadMap(this->_links[Input::LEFT]);
142  pos.x = this->_size.x * this->_tileSize - size.x / 2.;
143  this->_savePlayer();
144 
145  } else if (pos.y < size.y / -2.) {
146  this->_serialize("saves/map_" + std::to_string(this->_id) + ".sav");
147  this->_loadMap(this->_links[Input::UP]);
148  pos.y = this->_size.y * this->_tileSize - size.y / 2.;
149  this->_savePlayer();
150 
151  } else if (pos.x + size.x / 2. > this->_size.x * this->_tileSize) {
152  this->_serialize("saves/map_" + std::to_string(this->_id) + ".sav");
153  this->_loadMap(this->_links[Input::RIGHT]);
154  pos.x = size.x / -2.;
155  this->_savePlayer();
156 
157  } else if (pos.y + size.y / 2. > this->_size.y * this->_tileSize) {
158  this->_serialize("saves/map_" + std::to_string(this->_id) + ".sav");
159  this->_loadMap(this->_links[Input::DOWN]);
160  pos.y = size.y / -2.;
161  this->_savePlayer();
162 
163  }
164 
165  for (auto &trigger : this->_tpTriggers)
166  if (
167  -this->_tileSize < trigger.location.x && trigger.location.x < static_cast<int>(size.x + this->_tileSize) &&
168  -this->_tileSize < trigger.location.y && trigger.location.y < static_cast<int>(size.y + this->_tileSize)
169  ) {
170  this->_loadMap(trigger.mapId);
171  pos = sf::Vector2f(trigger.mapSpawn.x, trigger.mapSpawn.y);
172  }
173  }

209  void Map::_loadFromStream(std::istream &stream, bool loadEntities)
210  {
211  logger.debug("Loading map");
212 
213  _game.resources.playMusic(_readString(stream));
214 
215  this->_tileMap = _readString(stream);
216  logger.debug("Tilemap file is '" + this->_tileMap + "'");
217 
218  this->_tileSize = _readInteger<unsigned char>(stream);
219  logger.debug("Tilesize is " + std::to_string(this->_tileSize));
220 
221  this->_size.x = _readInteger<unsigned short>(stream);
222  this->_size.y = _readInteger<unsigned short>(stream);
223  logger.debug("Map size is " + Utils::toString(this->_size) + " tiles");
224 
225  for (auto len = _readInteger<unsigned>(stream); len != 0; len--) {
226  auto &result = this->_tpTriggers.emplace_back(stream);
227 
228  logger.debug(
229  std::to_string(len) + " teleporters remaining ->"
230  "Position: " + Utils::toString(result.location) +
231  "Map: " + std::to_string(result.mapId) +
232  "Spawn: " + Utils::toString(result.mapSpawn)
233  );
234  }
235 
236  std::memset(this->_links, 0, sizeof(this->_links));
237  for (unsigned i = 0; i < 4; i++) {
238  this->_links[i] = _readInteger<unsigned short>(stream);
239  logger.debug("Link " + Input::actionToString(static_cast<Input::Action>(i)) + " is to map " + std::to_string(this->_links[i]));
240  }
241 
242  logger.debug("Fetching objects");
243  for (unsigned y = 0; y < this->_size.y; y++)
244  for (unsigned x = 0; x < this->_size.x; x++)
245  this->_objects.push_back(_readInteger<unsigned char>(stream));
246 
247  if (loadEntities) {
248  logger.debug("Loading entities");
249  stream >> this->_core;
250 
251  if (this->_core.getEntityByName("Player").empty())
252  this->_core.makeEntity("Player");
253  }
254  logger.debug("Operation completed");
255  }

294  for (unsigned x = pos.x < 0 ? 0 : pos.x; x < pos.x + size.x && x < this->_size.x; x++) {
295  _game.resources.screen->draw(
