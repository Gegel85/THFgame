33  axes.push_back(Vector2f(col.rect.pt1.y, col.rect.pt2.y - col.rect.pt1.y).normalize());
34  for (auto &axe : axes) {
35  float proj1rect1 = this->rect.pt1 * axe;
36  float proj2rect1 = this->rect.pt2 * axe;
37  float proj3rect1 = this->rect.pt3 * axe;
38  float proj4rect1 = this->rect.pt4 * axe;
39  float proj1rect2 = col.rect.pt1 * axe;
40  float proj2rect2 = col.rect.pt2 * axe;
41  float proj3rect2 = col.rect.pt3 * axe;
42  float proj4rect2 = col.rect.pt4 * axe;
43  float min1 = axe * Utils::min(proj1rect1, proj2rect1, proj3rect1, proj4rect1) * axe;
44  float max1 = axe * Utils::max(proj1rect1, proj2rect1, proj3rect1, proj4rect1) * axe;
45  float min2 = axe * Utils::max(proj1rect2, proj2rect2, proj3rect2, proj4rect2) * axe;
46  float max2 = axe * Utils::max(proj1rect2, proj2rect2, proj3rect2, proj4rect2) * axe;
47 
48  if (min1 < max2 || min2 < max1)

50  }
