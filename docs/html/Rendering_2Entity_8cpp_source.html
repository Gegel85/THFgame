20  void Entity::Config::loadFile(Resources &resources, const std::string &path)
21  {
22  std::ifstream stream{path};
23  nlohmann::json value;
24 
25  if (stream.fail())
26  throw InvalidAnimationConfigException("Cannot open file " + path);
27 
28  try {
29  unsigned counter = 0;
30 
31  stream >> value;
32  this->texture = value["texture"];
33  this->tileSize = {value["tileSize"]["x"], value["tileSize"]["y"]};
34  for (auto &val : value["animations"]) {
35  if (val == 0)
36  throw InvalidAnimationConfigException("");
37  this->animations.push_back(val);
38  }
39  for (auto &val : value["delays"])
40  this->delays.push_back(val);
41  for (int i = 0; i < DEAD; i++)
42  for (int j = 0; j < 8; j++) {
43  this->animationStart.emplace_back(counter);
44  counter += this->animations[i];
45  }
46  this->animationStart.emplace_back(counter);
47  this->textureSize = resources.textures.at(this->texture).getSize();
48  if (this->textureSize.x % this->tileSize.x || this->textureSize.y % this->tileSize.y)
49  throw InvalidAnimationConfigException(
50  "The texture chosen (" + this->texture +
51  ") size isn't a perfect grid of " +
52  std::to_string(this->tileSize.x) + "x" + std::to_string(this->tileSize.y) +
53  " tiles"
54  );
55  } catch (std::out_of_range &) {
56  throw InvalidAnimationConfigException("The texture given is not loaded: " + this->texture);
57  } catch (nlohmann::detail::parse_error &e) {
58  throw InvalidAnimationConfigException("The JSON file has an invalid format: " + std::string(e.what()));
59  } catch (nlohmann::detail::type_error &e) {
60  throw InvalidAnimationConfigException("The JSON values are invalid: " + std::string(e.what()));
61  }
62  if (this->animations.size() != DEAD + 1)
63  throw InvalidAnimationConfigException(
64  "Expected " + std::to_string(DEAD + 1) + " animations but found " + std::to_string(this->animations.size())
65  );
66  if (this->delays.size() != DEAD + 1)
67  throw InvalidAnimationConfigException(
68  "Expected " + std::to_string(DEAD + 1) + " delays but found " + std::to_string(this->delays.size())
69  );
70  }

170  void Entity::update()
171  {
172  if (!this->_idleDelay) {
173  std::uniform_int_distribution distribution{0, 59};
174  std::uniform_int_distribution distribution2{0, 1};
175 
176  if (!distribution(this->_resources.random)) {
177  this->_animation = static_cast<Animation>(IDLEANIM1 + distribution2(this->_resources.random));
178  this->_animationState = 0;
179  this->_delay = this->_configs.delays[this->_animation];
180  }
181  this->_idleDelay = 20;
182  } else
183  this->_idleDelay -= this->_animation == IDLE;
184 
185  if (!this->_delay) {
186  this->_delay = this->_configs.delays[this->_animation];
187  switch (this->_animation) {
188  case IDLEANIM1:
189  case IDLEANIM2:
190  if (this->_animationState + 1 == this->_configs.animations[this->_animation])
191  this->_animation = IDLE;
192  break;
193  case DEAD:
194  case HIT:
195  if (this->_animationState + 2 == this->_configs.animations[this->_animation])
196  this->_delay = -1;
197  break;
198  default:
199  break;
200  }
201  this->_animationState = (this->_animationState + 1) % this->_configs.animations[this->_animation];
202  } else
203  this->_delay--;
204  }
205 }
