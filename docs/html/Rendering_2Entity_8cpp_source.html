28  for (auto &val : value["animations"]) {

34  this->delays.push_back(val);
35  for (int i = 0; i < DEAD; i++)
36  for (int j = 0; j < 8; j++) {
37  this->animationStart.emplace_back(counter);
38  counter += this->animations[i];
39  }
40  this->animationStart.emplace_back(counter);
41  this->textureSize = resources.textures.at(this->texture).getSize();
42  if (this->textureSize.x % this->tileSize.x || this->textureSize.y % this->tileSize.y)
43  throw InvalidAnimationConfigException(
44  "The texture chosen (" + this->texture +
45  ") size isn't a perfect grid of " +
46  std::to_string(this->tileSize.x) + "x" + std::to_string(this->tileSize.y) +
47  " tiles"
48  );
49  } catch (std::out_of_range &) {
50  throw InvalidAnimationConfigException("The texture given is not loaded: " + this->texture);
51  } catch (nlohmann::detail::parse_error &e) {
52  throw InvalidAnimationConfigException("The JSON file has an invalid format: " + std::string(e.what()));
53  } catch (nlohmann::detail::type_error &e) {
54  throw InvalidAnimationConfigException("The JSON values are invalid: " + std::string(e.what()));
55  }
56  if (this->animations.size() != DEAD + 1)

133  void Entity::update()
134  {
135  if (!this->_idleDelay) {
136  std::uniform_int_distribution distribution{0, 59};
137  std::uniform_int_distribution distribution2{0, 1};
138 
139  if (!distribution(this->_resources.random)) {
140  this->_animation = static_cast<Animation>(IDLEANIM1 + distribution2(this->_resources.random));
141  this->_animationState = 0;
142  this->_delay = this->_configs.delays[this->_animation];
143  }
144  this->_idleDelay = 20;
145  } else
146  this->_idleDelay -= this->_animation == IDLE;
147 
148  if (!this->_delay) {
149  this->_delay = this->_configs.delays[this->_animation];
150  switch (this->_animation) {
151  case IDLEANIM1:
152  case IDLEANIM2:
153  if (this->_animationState + 1 == this->_configs.animations[this->_animation])
154  this->_animation = IDLE;
155  break;
156  case DEAD:
157  case HIT:
158  if (this->_animationState + 2 == this->_configs.animations[this->_animation])
159  this->_delay = -1;
160  break;
161  default:
162  break;
163  }
164  this->_animationState = (this->_animationState + 1) % this->_configs.animations[this->_animation];
165  } else
166  this->_delay--;
167  }
