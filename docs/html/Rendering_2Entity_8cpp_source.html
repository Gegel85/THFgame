37  this->animationStart.emplace_back(counter);

40  this->animationStart.emplace_back(counter);
41  this->textureSize = _game.resources.textures.at(this->texture).getSize();
42  if (this->textureSize.x % this->tileSize.x || this->textureSize.y % this->tileSize.y)
43  throw InvalidAnimationConfigException(
44  "The texture chosen (" + this->texture +
45  ") size isn't a perfect grid of " +
46  std::to_string(this->tileSize.x) + "x" + std::to_string(this->tileSize.y) +
47  " tiles"
48  );
49  } catch (std::out_of_range &) {
50  throw InvalidAnimationConfigException("The texture given is not loaded: " + this->texture);
51  } catch (nlohmann::detail::parse_error &e) {

53  } catch (nlohmann::detail::type_error &e) {

132  void Entity::update()
133  {
134  if (!this->_idleDelay) {
135  std::uniform_int_distribution distribution{0, 59};
136  std::uniform_int_distribution distribution2{0, 1};
137 
138  if (!distribution(_game.resources.random)) {
139  this->_animation = static_cast<Animation>(IDLEANIM1 + distribution2(_game.resources.random));
140  this->_animationState = 0;
141  this->_delay = this->_configs.delays[this->_animation];
142  }
143  this->_idleDelay = 20;
144  } else
145  this->_idleDelay -= this->_animation == IDLE;
146 
147  if (!this->_delay) {
148  this->_delay = this->_configs.delays[this->_animation];
149  switch (this->_animation) {
150  case IDLEANIM1:
151  case IDLEANIM2:
152  if (this->_animationState + 1 == this->_configs.animations[this->_animation])
153  this->_animation = IDLE;
154  break;
155  case DEAD:
156  case HIT:
157  if (this->_animationState + 2 == this->_configs.animations[this->_animation])
158  this->_delay = -1;
159  break;
160  default:
161  break;
162  }
163  this->_animationState = (this->_animationState + 1) % this->_configs.animations[this->_animation];
164  } else
165  this->_delay--;
166  }
