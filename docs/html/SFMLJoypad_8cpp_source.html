38  void SFMLJoypad::handleEvent(sf::Event event)
39  {
40  if (!this->_keyChanging)
41  return;
42 
43  switch (*this->_keyChanging) {
44  case UP:
45  case DOWN:
46  if (event.type == sf::Event::JoystickMoved && event.joystickMove.joystickId == 0) {
47  this->_axis.y = event.joystickMove.axis;
48  this->_keyChanging.reset();
49  this->_keyChanged = true;
50  }
51  break;
52  case LEFT:
53  case RIGHT:
54  if (event.type == sf::Event::JoystickMoved && event.joystickMove.joystickId == 0) {
55  this->_axis.x = event.joystickMove.axis;
56  this->_keyChanging.reset();
57  this->_keyChanged = true;
58  }
59  break;
60  default:
61  if (event.type == sf::Event::JoystickButtonPressed && event.joystickButton.joystickId == 0) {
62  this->_buttons[*this->_keyChanging - 4] = event.joystickButton.button;
63  this->_keyChanging.reset();
64  this->_keyChanged = true;
65  }
66  }

69  double SFMLJoypad::getDirectionAngle() const
70  {
71  sf::Vector2f vec{
72  sf::Joystick::getAxisPosition(0, this->_axis.x),
73  sf::Joystick::getAxisPosition(0, this->_axis.y)
74  };
75 
76  if (std::abs(vec.x) <= ACTIVATION_THRESHOLD && std::abs(vec.y) <= ACTIVATION_THRESHOLD)
77  return 0;
78  else if (std::abs(vec.x) <= ACTIVATION_THRESHOLD)
79  return vec.y / 100 * M_PI_2;
80  else if (std::abs(vec.y) <= ACTIVATION_THRESHOLD)
81  return (vec.x - 100) / 200 * -M_PI;
82  else if (vec.x < 0 && vec.y < 0)
83  return std::atan(vec.y / vec.x) + M_PI;
84  else if (vec.x < 0)
85  return -std::atan(vec.x / vec.y) + M_PI - M_PI_2;
86  else if (vec.y < 0)
87  return -std::atan(vec.x / vec.y) - M_PI_2;
88  else
89  return std::atan(vec.y / vec.x);
90  }

92  bool SFMLJoypad::actionPressed(Action action) const
93  {
94  if (this->_window && !this->_window->hasFocus())
95  return false;
96  switch (action) {
97  case UP:
98  return sf::Joystick::getAxisPosition(0, this->_axis.x) < -ACTIVATION_THRESHOLD;
99  case DOWN:

101  case LEFT:

103  case RIGHT:
104  return sf::Joystick::getAxisPosition(0, this->_axis.y) > ACTIVATION_THRESHOLD;
105  default:
106  return sf::Joystick::isButtonPressed(0, this->_buttons[action - 4]);
107  }
