143  double SFMLKeyboard::getDirectionAngle() const
144  {
145  bool dirs[4] = {
146  this->actionPressed(UP),
147  this->actionPressed(DOWN),
148  this->actionPressed(LEFT),
149  this->actionPressed(RIGHT),
150  };
151 
152  reinterpret_cast<short *>(dirs)[0] *= dirs[0] != dirs[1];
153  reinterpret_cast<short *>(dirs)[1] *= dirs[2] != dirs[3];
154 
155  if (*reinterpret_cast<unsigned *>(dirs) == 0)
156  return 0;
157 
158  if (dirs[0] + dirs[1] + dirs[2] + dirs[3] == 1)
159  return (dirs[0] * -M_PI_2 + dirs[1] * M_PI_2 + dirs[2] * M_PI);
160  return (M_PI_2 + dirs[2] * M_PI_4 - dirs[3] * M_PI_4) * (dirs[1] - dirs[0]);
161  }

209  void SFMLKeyboard::_updateState()
210  {
211  if (this->_window && !this->_window->hasFocus())
212  return;
213  for (unsigned i = 0; i < NB_OF_ACTION; i++) {
214  bool press = this->actionPressed(static_cast<Action>(i));
215 
216  if (press != this->_state[i]) {
217  this->_state[i] = press;
218  this->_events.emplace_back(
219  press ? Event::EVENT_TRIGGERED : Event::EVENT_STOPPED_TRIGGER,
220  static_cast<Action>(i)
221  );
222  }
223  }
224  }

226  std::optional<Input::Event> SFMLKeyboard::pollEvent()
227  {
228  if (this->_events.empty())
229  this->_updateState();
230 
231  if (!this->_events.empty()) {
232  auto event = this->_events[0];
233 
234  this->_events.erase(this->_events.begin());
235  return event;
